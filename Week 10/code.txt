// Week-10 LINEAR SEARCH (Iterative)

public class LinearSearch {

    public static int linearSearch(int[] arr, int target) {
        for (int i = 0; i < arr.length; i++) {
            if (arr[i] == target) {
                return i;   // found
            }
        }
        return -1;  // not found
    }

    public static void main(String[] args) {
        int[] arr = {7, 14, 9, 28, 6};
        int index = linearSearch(arr, 28);

        System.out.println(index == -1 ? "Not Found" : "Found at index: " + index);
    }
}

/*
Output:
Found at index: 3
*/


// Week-10 BINARY SEARCH (Iterative)

public static int binarySearchIterative(int[] arr, int target) {

    int low = 0, high = arr.length - 1;

    while (low <= high) {

        int mid = (low + high) / 2;

        if (arr[mid] == target) return mid;

        else if (target > arr[mid])
            low = mid + 1;
        else
            high = mid - 1;
    }

    return -1;
}

/*
Example usage:
int[] arr = {2, 5, 8, 12, 16};
int index = binarySearchIterative(arr, 12);
Output → Found at index: 3
*/


// #Week-10 LINEAR SEARCH (Recursive)

public class LinearSearchRecursive {

    // Recursive function
    public static int linearSearchRec(int[] arr, int target, int index) {

        // base case: index out of range
        if (index == arr.length) {
            return -1;   // not found
        }

        // check current element
        if (arr[index] == target) {
            return index;
        }

        // recursive call for next index
        return linearSearchRec(arr, target, index + 1);
    }

    // Main function
    public static void main(String[] args) {

        int[] arr = {3, 8, 12, 19, 7};
        int target = 19;

        int result = linearSearchRec(arr, target, 0);

        if (result == -1)
            System.out.println("Element not found");
        else
            System.out.println("Found at index: " + result);
    }
}

/*
Output:
Found at index: 3
*/


// Week-10 BINARY SEARCH (Recursive)

public static int binarySearchRecursive(int[] arr, int low, int high, int target) {

    if (low > high) return -1;

    int mid = (low + high) / 2;

    if (arr[mid] == target) return mid;

    if (target < arr[mid])
        return binarySearchRecursive(arr, low, mid - 1, target);

    return binarySearchRecursive(arr, mid + 1, high, target);
}

/*
Example usage:
int[] arr = {1, 4, 7, 10, 13};
int index = binarySearchRecursive(arr, 0, arr.length - 1, 10);
Output → Found at index: 3
*/
